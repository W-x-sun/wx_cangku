69  GLM_FUNC_QUALIFIER detail::tmat3x3<T, P> reflect2D(
70  const detail::tmat3x3<T, P>& m,
71  const detail::tvec3<T, P>& normal)
72  {
73  detail::tmat3x3<T, P> r(1);
74  r[0][0] = 1 - 2 * normal.x * normal.x;
75  r[0][1] = -2 * normal.x * normal.y;
76  r[1][0] = -2 * normal.x * normal.y;
77  r[1][1] = 1 - 2 * normal.y * normal.y;
78  return m * r;
