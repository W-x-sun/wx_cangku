69  GLM_FUNC_QUALIFIER detail::tmat4x4<T, P> rotate
70  (
71  detail::tmat4x4<T, P> const & m,
72  T const & angle,
73  detail::tvec3<T, P> const & v
74  )
75  {
76 #ifdef GLM_FORCE_RADIANS
77  T a = angle;
78 #else
79 # pragma message("GLM: rotate function taking degrees as a parameter is deprecated. #define GLM_FORCE_RADIANS before including GLM headers to remove this message.")
80  T a = radians(angle);
81 #endif
82  T c = cos(a);
83  T s = sin(a);
84 
85  detail::tvec3<T, P> axis(normalize(v));
86  detail::tvec3<T, P> temp((T(1) - c) * axis);
87 
88  detail::tmat4x4<T, P> Rotate(detail::tmat4x4<T, P>::_null);
89  Rotate[0][0] = c + temp[0] * axis[0];
90  Rotate[0][1] = 0 + temp[0] * axis[1] + s * axis[2];
91  Rotate[0][2] = 0 + temp[0] * axis[2] - s * axis[1];
92 
93  Rotate[1][0] = 0 + temp[1] * axis[0] - s * axis[2];
94  Rotate[1][1] = c + temp[1] * axis[1];
95  Rotate[1][2] = 0 + temp[1] * axis[2] + s * axis[0];
96 
97  Rotate[2][0] = 0 + temp[2] * axis[0] + s * axis[1];
98  Rotate[2][1] = 0 + temp[2] * axis[1] - s * axis[0];
99  Rotate[2][2] = c + temp[2] * axis[2];
100 
101  detail::tmat4x4<T, P> Result(detail::tmat4x4<T, P>::_null);
102  Result[0] = m[0] * Rotate[0][0] + m[1] * Rotate[0][1] + m[2] * Rotate[0][2];
103  Result[1] = m[0] * Rotate[1][0] + m[1] * Rotate[1][1] + m[2] * Rotate[1][2];
104  Result[2] = m[0] * Rotate[2][0] + m[1] * Rotate[2][1] + m[2] * Rotate[2][2];

111  (
112  detail::tmat4x4<T, P> const & m,
113  T const & angle,
114  detail::tvec3<T, P> const & v
115  )
116  {

118  T const a = angle;
119 #else

121  T const a = radians(angle);

124  T s = sin(a);
125  detail::tmat4x4<T, P> Result;
126 
127  detail::tvec3<T, P> axis = normalize(v);
128 
129  Result[0][0] = c + (1 - c) * axis.x * axis.x;
130  Result[0][1] = (1 - c) * axis.x * axis.y + s * axis.z;
131  Result[0][2] = (1 - c) * axis.x * axis.z - s * axis.y;
132  Result[0][3] = 0;

135  Result[1][1] = c + (1 - c) * axis.y * axis.y;
136  Result[1][2] = (1 - c) * axis.y * axis.z + s * axis.x;
137  Result[1][3] = 0;

140  Result[2][1] = (1 - c) * axis.z * axis.y - s * axis.x;
141  Result[2][2] = c + (1 - c) * axis.z * axis.z;
142  Result[2][3] = 0;
143 
144  Result[3] = detail::tvec4<T, P>(0, 0, 0, 1);

239  GLM_FUNC_QUALIFIER detail::tmat4x4<valType, defaultp> perspective
240  (
241  valType const & fovy,
242  valType const & aspect,
243  valType const & zNear,
244  valType const & zFar
245  )
246  {
247  assert(aspect != valType(0));
248  assert(zFar != zNear);

257  valType tanHalfFovy = tan(rad / valType(2));

260  Result[0][0] = valType(1) / (aspect * tanHalfFovy);

262  Result[2][2] = - (zFar + zNear) / (zFar - zNear);

356  template <typename T, typename U, precision P>
357  GLM_FUNC_QUALIFIER detail::tvec3<T, P> project
358  (
359  detail::tvec3<T, P> const & obj,
360  detail::tmat4x4<T, P> const & model,
361  detail::tmat4x4<T, P> const & proj,
362  detail::tvec4<U, P> const & viewport
363  )
364  {
365  detail::tvec4<T, P> tmp = detail::tvec4<T, P>(obj, T(1));
366  tmp = model * tmp;
367  tmp = proj * tmp;
368

372  tmp[1] = tmp[1] * T(viewport[3]) + T(viewport[1]);
373 
374  return detail::tvec3<T, P>(tmp);
