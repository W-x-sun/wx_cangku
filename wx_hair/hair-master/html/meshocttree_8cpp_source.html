97 bool MeshOctTree::rayIntersectsCube(glm::vec3 pt, glm::vec3 dir, glm::vec3 minCorner, glm::vec3 maxCorner){
98  glm::vec3 recipDir;
99  // r.dir is unit direction vector of ray
100  recipDir.x = 1.0f / dir.x;
101  recipDir.y = 1.0f / dir.y;
102  recipDir.z = 1.0f / dir.z;
103  // lb is the corner of AABB with minimal coordinates - left bottom, rt is maximal corner
104  // r.org is origin of ray
105  float t1 = (minCorner.x - pt.x)*recipDir.x;
106  float t2 = (maxCorner.x - pt.x)*recipDir.x;
107  float t3 = (minCorner.y - pt.y)*recipDir.y;
108  float t4 = (maxCorner.y - pt.y)*recipDir.y;
109  float t5 = (minCorner.z - pt.z)*recipDir.z;
110  float t6 = (maxCorner.z - pt.z)*recipDir.z;
111 
112  float tmin = max(max(min(t1, t2), min(t3, t4)), min(t5, t6));
113  float tmax = min(min(max(t1, t2), max(t3, t4)), max(t5, t6));
114 
115  // if tmax < 0, ray (line) is intersecting AABB, but whole AABB is behing us
116  if (tmax < 0) return false;
117 
118  // if tmin > tmax, ray doesn't intersect AABB
119  if (tmin > tmax) return false;
120 
121  return true;
122 }
