83  // Using Moller-Trumbore method for triangle-ray intersection
84  bool intersect(glm::vec3 &intersection, glm::vec3 ro, glm::vec3 rd)
85  {
86  glm::vec3 edge1 = v2 - v1;
87  glm::vec3 edge2 = v3 - v1;
88  glm::vec3 pVec = glm::cross(rd, edge2);
89 
90  float det = glm::dot(edge1, pVec);
91  if (det == 0) return false;
92  float invDet = 1.0f / det;
93 
94  glm::vec3 tVec = ro - v1;
95  float u = glm::dot(tVec, pVec) * invDet;
96  if (u < 0.0 || u > 1.0) return false;
97 
98  glm::vec3 qVec = glm::cross(tVec, edge1);
99  float v = glm::dot(rd, qVec) * invDet;
100  if (v < 0.0 || u + v > 1.0) return false;
101 
102  glm::vec3 intersectionPoint = v1 + u * edge1 + v * edge2;
103  float t = (intersectionPoint.x - ro.x) / rd.x;
104  if (t < 0) return false;
105 
106  intersection = intersectionPoint;
107  return true;
108  }

112 inline unsigned int htoi(const char s[])
113 {
114  unsigned int val = 0;
115  int x = 0;
116 
117  if(s[x] == '0' && (s[x+1]=='x' || s[x+1]=='X')) x+=2;
118 
119  while(s[x]!='\0')
120  {
121  if(val > UINT_MAX) return 0;
122  else if(s[x] >= '0' && s[x] <='9')
123  {
124  val = val * 16 + s[x] - '0';
125  }

128  val = val * 16 + s[x] - 'A' + 10;

133  }
