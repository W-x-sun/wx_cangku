298  // refract
299  template <typename genType>
300  GLM_FUNC_QUALIFIER genType refract
301  (
302  genType const & I,
303  genType const & N,
304  genType const & eta
305  )
306  {
307  GLM_STATIC_ASSERT(std::numeric_limits<genType>::is_iec559, "'refract' only accept floating-point inputs");
308 
309  genType dotValue = dot(N, I);
310  genType k = genType(1) - eta * eta * (genType(1) - dotValue * dotValue);
311  if(k < genType(0))
312  return genType(0);
313  else
314  return eta * I - (eta * dotValue + sqrt(k)) * N;
315  }

317  template <typename T, precision P, template <typename, precision> class vecType>
318  GLM_FUNC_QUALIFIER vecType<T, P> refract
319  (
320  vecType<T, P> const & I,
321  vecType<T, P> const & N,
322  T const & eta
323  )
324  {
325  GLM_STATIC_ASSERT(std::numeric_limits<T>::is_iec559, "'refract' only accept floating-point inputs");
326 
327  T dotValue = dot(N, I);
328  T k = T(1) - eta * eta * (T(1) - dotValue * dotValue);
329  if(k < T(0))
330  return vecType<T, P>(0);
331  else
332  return eta * I - (eta * dotValue + std::sqrt(k)) * N;
333  }
