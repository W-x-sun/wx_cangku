102  template <typename T, precision P, template <typename, precision> class vecType>
103  GLM_FUNC_QUALIFIER bool areOrthogonal
104  (
105  vecType<T, P> const & v0,
106  vecType<T, P> const & v1,
107  T const & epsilon
108  )
109  {
110  GLM_STATIC_ASSERT(std::numeric_limits<T>::is_iec559, "'areOrthogonal' only accept floating-point inputs");
111 
112  return abs(dot(v0, v1)) <= max(
113  static_cast<T>(1),
114  length(v0)) * max(static_cast<T>(1), length(v1)) * epsilon;
115  }
