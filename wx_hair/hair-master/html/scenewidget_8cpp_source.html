153 //纹理绘制函数（参数：坐标、纹理的image.bits()、尺寸）
154 void SceneWidget::paintTexture(glm::vec2 center, uchar *data, glm::vec2 imgSize)
155 {
156  int canvasW = imgSize.x;
157  int canvasH = imgSize.y;
158 
159  int maskSize = 2*m_radius+1;
160 
161  int colStart = center.x-m_radius;
162  int colEnd = center.x+m_radius+1;
163  int rowStart = center.y-m_radius;
164  int rowEnd = center.y+m_radius+1;
165 
166  int maskColStart = 0;
167  int maskColEnd = maskSize;
168  int maskRowStart = 0;
169  int maskRowEnd = maskSize;
170 
171  if (colStart < 0){
172  maskColStart -= colStart;
173  colStart = 0;
174  }
175  if (colEnd > canvasW){
176  maskColEnd -= (colEnd-canvasW);
177  colEnd = canvasW;
178  }
179  if (rowStart < 0){
180  maskRowStart -= rowStart;
181  rowStart = 0;
182  }
183  if (rowEnd > canvasH){
184  maskRowEnd -= (rowEnd-canvasH);
185  rowEnd = canvasH;
186  }
187 
188  BGRA *pix = (BGRA*)data;
189 
190  if (!m_blendBufferValid){
191  m_blendBufferValid = true;
192 
193  delete[] m_blendBuffer;
194  m_blendBuffer = new float[canvasW*canvasH];
195  int i; for (i = 0; i < canvasW*canvasH; i++) m_blendBuffer[i] = 0;
196  }
197 
198  int row, col, maskRow, maskCol;
199  for (row = rowStart, maskRow = maskRowStart; row < rowEnd; row++, maskRow++){
200  for (col = colStart, maskCol = maskColStart; col < colEnd; col++, maskCol++){
201  float maskValue = m_mask[maskRow*maskSize+maskCol];
202 
203  float blendValue = m_opacity*maskValue;
204 
205  float blendLeftToMax = (m_opacity-m_blendBuffer[row*canvasW+col])/(1.0-m_blendBuffer[row*canvasW+col]);
206  if (blendValue > blendLeftToMax) blendValue = blendLeftToMax;
207  if (blendValue < 0) blendValue = 0;
208  m_blendBuffer[row*canvasW+col] += (1.0-m_blendBuffer[row*canvasW+col])*blendValue; // incorporate previous blends
209 
210  if (m_currentTexture == m_densityMapTexture){
211  pix[row*canvasW+col].r = (pix[row*canvasW+col].r)*(1-blendValue) + (255*m_grayscale)*(blendValue);
212  pix[row*canvasW+col].g = (pix[row*canvasW+col].g)*(1-blendValue) + (255*m_grayscale)*(blendValue);
213  pix[row*canvasW+col].b = (pix[row*canvasW+col].b)*(1-blendValue) + (255*m_grayscale)*(blendValue);
214  } else if (m_currentTexture == m_directionMapTexture){
215  if (m_brushDir == "left" || m_brushDir == "right"){
216  pix[row*canvasW+col].r = (pix[row*canvasW+col].r)*(1-blendValue) + (255*m_brushDirColor.x)*(blendValue);
217  } else if (m_brushDir == "up" || m_brushDir == "down"){
218  pix[row*canvasW+col].g = (pix[row*canvasW+col].g)*(1-blendValue) + (255*m_brushDirColor.y)*(blendValue);
219  }
220  }
221  }
222  }
223 }

234 //刷子设置（持续不变）对应实现
235 void SceneWidget::makeConstantMask()
236 {
237  delete[] m_mask;
238  int len = (2*m_radius+1);
239  m_mask = new float[len*len];//根据刷子半径生成mask（2r+1近似创建了一个圆形mask）
240 
241  int row, col;
242  for (row = 0; row < len; row++){
243  for (col = 0; col < len; col++){
244  float distance = sqrt((row-m_radius)*(row-m_radius)+(col-m_radius)*(col-m_radius));
245  m_mask[row*len+col] = (distance <= m_radius ? 1 : 0);
246  }
247  }
248 }
249 //刷子设置（线性）对应实现
250 void SceneWidget::makeLinearMask()
251 {
252  delete[] m_mask;
253  int len = (2*m_radius+1);
254  m_mask = new float[len*len];
255 
256  int row, col;
257  for (row = 0; row < len; row++){
258  for (col = 0; col < len; col++){
259  float distance = sqrt((row-m_radius)*(row-m_radius)+(col-m_radius)*(col-m_radius));
260  m_mask[row*len+col] = 1-distance/m_radius;
261  if (m_mask[row*len+col] < 0) m_mask[row*len+col] = 0;
262  }
263  }

270  m_mask = new float[len*len];
271 
272  int row, col;
273  for (row = 0; row < len; row++){
274  for (col = 0; col < len; col++){
275  float distance = sqrt((row-m_radius)*(row-m_radius)+(col-m_radius)*(col-m_radius));
276  m_mask[row*len+col] = (1-distance/m_radius);
277  if (m_mask[row*len+col] < 0) m_mask[row*len+col] = 0; // because apparently using max(a, b) is a bitch
